pukara_advance = {
	age = age_1_traditions
	potential = {
		OR = {
			AND = {
				has_variable = hafp_enabled
				OR = {
					has_variable = hafp_all_advances_enabled
					has_variable = hafp_g_age_age_1_traditions
					has_variable = hafp_g_f_culture_andean
				}
			}
			AND = {
				culture = { has_culture_group = culture_group:andean_group }
			}
		}
	}
	unlock_building = pukara_building
	requires = city_building_advance
}
