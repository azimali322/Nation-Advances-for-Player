griot_advance = {
	potential = {
		OR = {
			AND = {
				has_variable = hafp_enabled
				OR = {
					has_variable = hafp_all_advances_enabled
					has_variable = hafp_g_age_age_1_traditions
					has_variable = hafp_g_f_culture_west_african
				}
			}
			AND = {
				culture = { has_culture_group = culture_group:west_african_group }
			}
		}
	}

	age = age_1_traditions
	num_possible_artists = 1
	set_cabinet_member_cost_modifier = -0.1
}
