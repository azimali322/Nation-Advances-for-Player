the_sahel_trade = {
	age = age_1_traditions
	icon = trade_caravans

	potential = {
		OR = {
			AND = {
				has_variable = hafp_enabled
				OR = {
					has_variable = hafp_all_advances_enabled
					has_variable = hafp_g_age_age_1_traditions
					has_variable = hafp_g_cont_africa
					has_variable = hafp_g_reg_guinea_region
					has_variable = hafp_g_reg_maghreb_region
					has_variable = hafp_g_reg_sahel_region
				}
			}
			AND = {
				original_capital ?= {
				OR = {
				sub_continent = sub_continent:west_africa
				region = region:maghreb_region
				}
				}
			}
		}
	}

	requires = slave_raiding_wars
	trade_land_efficiency = medium_trade_land_efficiency_bonus
	allow_tributary_subject = yes
}
