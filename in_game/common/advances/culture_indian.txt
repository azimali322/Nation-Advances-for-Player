paik_system_advance = {
	age = age_1_traditions
	potential = {
		OR = {
			AND = {
				has_variable = hafp_enabled
				OR = {
					has_variable = hafp_all_advances_enabled
					has_variable = hafp_g_age_age_1_traditions
					has_variable = hafp_g_f_culture_indian
				}
			}
			AND = {
				OR = {
				has_or_had_tag = AHO
				has_or_had_tag = MLB
				AND = {
				culture = { has_culture_group = culture_group:indian_group }
				original_capital ?= {
				OR = {
				area = area:bengal_area
				area = area:assam_area
				}
				}
				}
				}
			}
		}
	}
	unlock_levy = levy_a_early_paik
	requires = unlock_feudal_levy_advance
}
