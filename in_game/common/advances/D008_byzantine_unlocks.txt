					has_variable = hafp_g_cont_asia
					has_variable = hafp_g_cont_europe
					has_variable = hafp_g_reg_anatolia_region
					has_variable = hafp_g_reg_balkan_region
					has_variable = hafp_g_reg_carpathia_region
					has_variable = hafp_g_reg_caucasus_region
					has_variable = hafp_g_reg_crescent_region
					has_variable = hafp_g_reg_egypt_region
					has_variable = hafp_g_reg_france_region
					has_variable = hafp_g_reg_great_britain_region
					has_variable = hafp_g_reg_iberia_region
					has_variable = hafp_g_reg_italy_region
					has_variable = hafp_g_reg_maghreb_region
					has_variable = hafp_g_reg_south_german_region
