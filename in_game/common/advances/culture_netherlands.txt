flemish_cloth_making = {
	age = age_1_traditions
	potential = {
		OR = {
			AND = {
				has_variable = hafp_enabled
				OR = {
					has_variable = hafp_all_advances_enabled
					has_variable = hafp_g_age_age_1_traditions
					has_variable = hafp_g_f_culture_netherlands
				}
			}
			AND = {
				culture = { has_culture_group = culture_group:netherlandish_group }
			}
		}
	}

	unlock_town_rights = flemish_cloth_industries_right
	
	requires = distribution_of_power_advance
}
