hafp_enable_all_advances = {

	on_own_nation = yes

	message = no

	potential = {
		scope:actor = {
			is_ai = no
			NOT = { has_variable = hafp_all_advances_enabled }
		}
	}

	allow = {
		scope:actor = {
			is_ai = no
			NOT = { has_variable = hafp_all_advances_enabled }
		}
	}

	select_trigger = {
		looking_for_a = character
		source = actor
		target_flag = recipient
		name = "choose_character"
		column = {
			data = name
		}
		visible = {
			is_alive = yes
			scope:actor = {
				NOT = { has_variable = hafp_all_advances_enabled }
			}
		}
		enabled = {
			is_ai = no
		}
	}

	ai_will_do = {
		add = {
			value = -1000
		}
	}

	effect = {
		scope:actor = {
			set_variable = hafp_all_advances_enabled
		}
	}
}
